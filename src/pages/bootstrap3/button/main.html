Button按钮

1. bs-style: 指定按钮的样式


    <!-- Standard button -->
    <button type="button" bs-style="default">（默认样式）Default</button>

    <!-- Provides extra visual weight and identifies the primary action in a set of buttons -->
    <button type="button" bs-style="primary">（首选项）Primary</button>

    <!-- Indicates a successful or positive action -->
    <button type="button" bs-style="success">（成功）Success</button>

    <!-- Contextual button for informational alert messages -->
    <button type="button" bs-style="info">（一般信息）Info</button>

    <!-- Indicates caution should be taken with this action -->
    <button type="button" bs-style="warning">（警告）Warning</button>

    <!-- Indicates a dangerous or potentially negative action -->
    <button type="button" bs-style="danger">（危险）Danger</button>

    <!-- Deemphasize a button by making it look like a link while maintaining button behavior -->
    <button type="button" bs-style="link">（链接）Link</button>
